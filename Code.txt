INPUT_SHAPE = (96, 96, 3)


def single_channel_model(shape=(96, 96, 1)):
    inputs = Input(shape=shape)
    x = Conv2D(8, 3, activation='relu', padding='same')(inputs)
    x = MaxPooling2D((2, 2))(x)

    x = SeparableConv2D(24, 3, activation='relu', padding='same')(x)
    x = MaxPooling2D((2, 2))(x)

    x = SeparableConv2D(48, 3, activation='relu', padding='same')(x)
    x = MaxPooling2D((2, 2))(x)

    x = SeparableConv2D(48, 3, activation='relu', padding='same')(x)
    x = MaxPooling2D((2, 2))(x)
    
    x = SeparableConv2D(48, 3, activation='relu', padding='same')(x)
    x = MaxPooling2D((2, 2))(x)

    return tf.keras.Model(inputs=inputs, outputs=x)
